#Prerequisitos
    Paquetes: python, node, npm, jdk8

#Instalar Android Studio

    Instalar Android Studio
    En la configuración del SDK de Android Estudio verificar:
        Google APIs, Android 23
        Android SDK Platform 23
        Intel x86 Atom_64 System Image
        Google APIs Intel x86 Atom_64 System Image
        Intel HAXM y vt-x de computadora activado
        Configurar desde Android Studio un dispositivo virtual

    #Añadir carpetas con dependencias de android a la variable PATH
    setx PATH = %PATH%;C:\Users\EliteBook\AppData\Local\Android\Sdk\tools
    setx PATH = %PATH%;C:\Users\EliteBook\AppData\Local\Android\Sdk\platform-tools
     
    abd devices
    android list avd ##android virtual devices
    android create avd -n avd-name --
    emulator -avd avd-name

#Crear proyecto de React Native
    #CLI de React Native
    npm install -g react-native-cli
    
    #Iniciar proyecto
    react-native init NombreAplicacion
    
    #Entrar a definición de proyecto
    cd NombreAplicacion
    
    #Abrir editor de proyecto
    code .
    
    #En otra ventana de instrucciones
    cd C:\Users\EliteBook\AppData\Local\Android\Sdk\tools
    
    #Lista de dispositivos virtuales
    android list avd
    
    #Emulador de dispositivo android
    emulator -avd NombreDispositivo
